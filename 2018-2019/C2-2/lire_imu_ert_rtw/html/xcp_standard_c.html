29static boolean_T isValidUploadSize(uint8_T size)



30{



31    size_t addressGranularity = XCP_ADDRESS_GRANULARITY_BYTES_NUMBER;



32    size_t maxCtoSize = xcpTransportMaxCtoSize();



33



34    return (size < ((maxCtoSize) / addressGranularity));



35}

68static XcpProtoErrorCode connectInputPacketHandler(void   *msgBuffer,



69                                                   size_t  xcpPacketOffset,



70                                                   size_t *outputPacketSize)



71{



72    XcpProtoErrorCode protoErrorCode = XCP_PROTO_SUCCESS;



73    uint8_T *packet = (uint8_T *)msgBuffer + xcpPacketOffset;



74    XcpConnectCmdPacketFrame *frame = (XcpConnectCmdPacketFrame *) packet;



75    boolean_T ok = false;



76



77    /* If DAQ List support is enabled, reset the status of dynamic DAQ list data structures */



78    ok = xcpResetDaqListStatus();



79



80    XCP_INPUT_PKT_ERROR_IF(!ok, XCP_PROTO_BUSY,



81        ("CONNECT: DAQ lists reset cannot be carried out, as an event was processed\n"));



82



83    /* Validate command inputs */



84    ok = (frame->mode == XCP_CONNECT_MODE_NORMAL) || (frame->mode == XCP_CONNECT_MODE_USER_DEFINED);



85    XCP_INPUT_PKT_ERROR_IF(!ok, XCP_PROTO_OUT_OF_RANGE,



86                          ("CONNECT: invalid mode detected %xH\n", frame->mode));



87



88    /* Everything is fine -> updating the Protocol Layer status */



89    XCP_PRINTF("CONNECT: entering XCP_CONNECTED status\n");



90    xcpStatusSet(XCP_CONNECTED);



91



92    protoErrorCode = XCP_PROTO_SUCCESS;



93    *outputPacketSize = XCP_CONNECT_RES_PACKET_SIZE_IN_BYTES;



94



95    return protoErrorCode;



96}

98static void connectOutputPacketHandler(XcpProtoErrorCode inputCode, void *packet, size_t packetSize)



99{



100    if (inputCode == XCP_PROTO_SUCCESS) {



101        /* Fill connect response */



102        XcpConnectResPacketFrame *frame = (XcpConnectResPacketFrame *)packet;



103        XCP_MEMSET(frame, 0, sizeof(*frame));



104



105        frame->PID = XCP_PID_RES;



106



107        /* Update resource field */



108        frame->resource = xcpResourceValue;



109



110        /* Update commModeBasic field */



111        frame->commModeBasic = xcpCommModeBaseValue;



112



113        /* Update Max CTO and DTO fields */



114        frame->maxCtoSize = (uint8_T)xcpTransportMaxCtoSize();



115        frame->maxDtoSize = (uint16_T)xcpTransportMaxDtoSize();



116



117        /* Update Protocol and Transport Version fields */



118        frame->xcpProtocolVersion  = XCP_MAJOR_NUMBER(XCP_PROTOCOL_LAYER_VERSION);



119        frame->xcpTransportVersion = XCP_MAJOR_NUMBER(XCP_TRANSPORT_LAYER_VERSION);



120



121        XCP_PRINTF("* Resource:          %xH\n", frame->resource);



122        XCP_PRINTF("* Comm Mode Basic:   %xH\n", frame->commModeBasic);



123        XCP_PRINTF("* Max CTO size:      %d\n",  frame->maxCtoSize);



124        XCP_PRINTF("* Max DTO size:      %d\n",  frame->maxDtoSize);



125        XCP_PRINTF("* Protocol Version:  %d\n",  frame->xcpProtocolVersion);



126        XCP_PRINTF("* Transport Version: %d\n",  frame->xcpTransportVersion);



127    }



128    else {



129        genericOutputPacketHandler(inputCode, packet, packetSize);



130    }



131}

136static XcpProtoErrorCode disconnectInputPacketHandler(void   *msgBuffer,



137                                                      size_t  xcpPacketOffset,



138                                                      size_t *outputPacketSize)



139{



140    boolean_T ok = true;



141



142    XCP_UNUSED_PARAM(msgBuffer);



143    XCP_UNUSED_PARAM(xcpPacketOffset);



144



145    /* If DAQ List support is enabled, reset the status of dynamic DAQ list data structures */



146    ok = xcpResetDaqListStatus();



147



148    XCP_INPUT_PKT_ERROR_IF(!ok, XCP_PROTO_BUSY,



149        ("DISCONNECT: DAQ lists reset cannot be carried out, as an event was processed\n"));



150



151    /* Everything is fine -> proceed with the status update */



152    XCP_PRINTF("DISCONNECT: entering XCP_DISCONNECTED status\n");



153    xcpStatusSet(XCP_DISCONNECTED);



154



155    *outputPacketSize = XCP_GENERIC_RES_PACKET_SIZE_IN_BYTES;



156



157    return XCP_PROTO_SUCCESS;



158}

194    XCP_PRINTF("* Session Config Id:           %d\n", frame->sessionConfigurationId);

248    static XcpProtoErrorCode uploadInputPacketHandler(void   *msgBuffer,



249        size_t  xcpPacketOffset,



250        size_t *outputPacketSize) {



251



252        XcpProtoErrorCode protoErrorCode = XCP_PROTO_SUCCESS;



253        uint8_T *packet = (uint8_T *)msgBuffer + xcpPacketOffset;



254        XcpUploadCmdPacketFrame *frame = (XcpUploadCmdPacketFrame *)packet;



255        boolean_T ok = false;



256



257        /* Check if the number of data elements is valid */



258        



259        ok = isValidUploadSize((uint8_T)frame->size);



260        XCP_INPUT_PKT_ERROR_IF(!ok, XCP_PROTO_OUT_OF_RANGE,



261            ("UPLOAD: invalid number of data elements (%d)\n", (uint8_T)frame->size));



262        



263        xcpUploadSize = (uint8_T)frame->size;



264        *outputPacketSize = XCP_UPLOAD_RES_PACKET_SIZE_IN_BYTES + XCP_IN_BYTES(xcpUploadSize);



265



266        return protoErrorCode;



267    }

270    static void uploadOutputPacketHandler(XcpProtoErrorCode inputCode, void *packet, size_t packetSize) {



271        



272        if (inputCode == XCP_PROTO_SUCCESS) {



273



274            /* Fill Upload response */



275            uint8_T *xcpMTARawPointer = xcpStandardGetAddressFromMta();



276            XcpUploadResPacketFrame *frame = (XcpUploadResPacketFrame *)packet;



277            XCP_MEMSET(frame, 0, sizeof(*frame));



278            frame->PID = XCP_PID_RES;



279



280            if ((xcpMTARawPointer != NULL) &&



281                (packetSize == (XCP_UPLOAD_RES_PACKET_SIZE_IN_BYTES + XCP_IN_BYTES(xcpUploadSize)))



282                ) {



283                uint8_T *pktData = (uint8_T *)packet + XCP_IN_AG(XCP_UPLOAD_RES_PACKET_SIZE_IN_BYTES);



284



285                XCP_PRINTF("UPLOAD: reading at address %p\n", xcpMTARawPointer);



286



287                /* Copy the memory content to the packet data area */



288                XCP_MEMCPY(pktData, xcpMTARawPointer, xcpUploadSize);



289



290                /* Post-increment MTA by the frame size in address granularity units */



291                xcpStandardIncrementMta(xcpUploadSize);



292                xcpUploadSize = 0;



293            }



294            else {



295                /* This is a software error and it should never happen */



296                XCP_PRINTF("UPLOAD: invalid data detected\n");



297            }



298        }



299        else {



300            genericOutputPacketHandler(inputCode, packet, packetSize);



301        }



302    }



303#endif  /* XCP_SET_MTA_ENABLE == 1  */

397uint8_T* xcpStandardGetAddressFromMta(void) {



398    return XCP_ADDRESS_GET(xcpMta.addressExtension, xcpMta.address);

401boolean_T xcpStandardSetMta(uint32_T address, uint8_T addressExtension) {



402    if ((addressExtension == 0) && (address == 0)) {



403        return false;



404    } else {



405        xcpMta.address = address;



406        xcpMta.addressExtension = addressExtension;



407        return true;



408    }



409}
