20 /** Transport Layer initialization parameters 



21     @note the default values can be overwritten when parsing the



22           External Mode command line arguments */



23static const void* xcpTransportLayerParams[] =



24{



25    "-port", "1",



26    "-baud", "57600",



27    "-verbose", "0",



28};

32void xcpExtModeParseArgs(int_T argc, const char_T *argv[])



33{



34#ifdef  EXTMODE_DISABLE_ARGS_PROCESSING



35    /* Some targets do not support command line args */



36    UNUSED_PARAMETER(argc);



37    UNUSED_PARAMETER(argv);



38#else



39    if ((argv != NULL) && (argc > 0)) {



40        int_T optionId = 1;



41        uint_T idx;



42



43        while (optionId < argc) {



44            const char_T *option = argv[optionId];



45



46            optionId++;



47



48            if ((option != NULL) && (optionId != argc)) {



49                boolean_T isXcpOption = false;



50



51                for (idx = 0; idx < sizeof(xcpTransportLayerParams)/sizeof(xcpTransportLayerParams[0]); idx = idx + 2){



52                    if (strcmp(option, xcpTransportLayerParams[idx]) == 0) {



53                        xcpTransportLayerParams[idx+1] = argv[optionId];



54                        isXcpOption = true;



55                        break;



56                    }



57                }



58



59                if (isXcpOption) {



60                    /* Mark arguments as "processed" */



61                    argv[optionId - 1] = NULL;



62                    argv[optionId] = NULL;



63                }



64                



65                optionId++;



66            }



67        }



68    }



69#endif



70}

72void xcpTransportGetInitParameters(int_T *parNumber, void **parList[])



73{



74#ifdef  EXTMODE_DISABLE_ARGS_PROCESSING



75    if ((parNumber != NULL) && (parList != NULL)) {



76        /* No Transport Layer parameters defined (assuming rtiostream default) */



77        *parNumber = 0;



78        *parList = NULL;



79    }



80#else    



81    if ((parNumber != NULL) && (parList != NULL)) {



82        *parNumber = XCP_ELEMENTS_NUMBER(xcpTransportLayerParams);



83        *parList = (void **) &xcpTransportLayerParams;



84    }



85#endif



86}



87



88void xcpGetInitParameters(int_T *parNumber, void **parList[])



89{



90    if ((parNumber != NULL) && (parList != NULL)) {



91        /* No Protocol Layer parameters available at the moment */



92        *parNumber = 0;



93        *parList = NULL;



94    }



95}
