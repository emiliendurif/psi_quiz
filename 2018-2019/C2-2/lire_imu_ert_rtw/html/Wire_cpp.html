29TwoWire::TwoWire(SERCOM * s, uint8_t pinSDA, uint8_t pinSCL)



30{



31  this->sercom = s;



32  this->_uc_pinSDA=pinSDA;



33  this->_uc_pinSCL=pinSCL;



34  transmissionBegun = false;



35}

37void TwoWire::begin(void) {



38  //Master Mode



39  sercom->initMasterWIRE(TWI_CLOCK);



40  sercom->enableWIRE();



41



42  pinPeripheral(_uc_pinSDA, g_APinDescription[_uc_pinSDA].ulPinType);



43  pinPeripheral(_uc_pinSCL, g_APinDescription[_uc_pinSCL].ulPinType);



44}

46void TwoWire::begin(uint8_t address, bool enableGeneralCall) {



47  //Slave mode

49  sercom->enableWIRE();



50



51  pinPeripheral(_uc_pinSDA, g_APinDescription[_uc_pinSDA].ulPinType);

65uint8_t TwoWire::requestFrom(uint8_t address, size_t quantity, bool stopBit)



66{



67  if(quantity == 0)



68  {



69    return 0;



70  }



71



72  size_t byteRead = 0;



73



74  rxBuffer.clear();



75



76  if(sercom->startTransmissionWIRE(address, WIRE_READ_FLAG))



77  {



78    // Read first data



79    rxBuffer.store_char(sercom->readDataWIRE());



80



81    // Connected to slave



82    for (byteRead = 1; byteRead < quantity; ++byteRead)



83    {



84      sercom->prepareAckBitWIRE();                          // Prepare Acknowledge



85      sercom->prepareCommandBitsWire(WIRE_MASTER_ACT_READ); // Prepare the ACK command for the slave



86      rxBuffer.store_char(sercom->readDataWIRE());          // Read data and send the ACK



87    }



88    sercom->prepareNackBitWIRE();                           // Prepare NACK to stop slave transmission



89    //sercom->readDataWIRE();                               // Clear data register to send NACK



90



91    if (stopBit)



92    {



93      sercom->prepareCommandBitsWire(WIRE_MASTER_ACT_STOP);   // Send Stop



94    }



95  }



96



97  return byteRead;



98}



99



100uint8_t TwoWire::requestFrom(uint8_t address, size_t quantity)



101{



102  return requestFrom(address, quantity, true);



103}

105void TwoWire::beginTransmission(uint8_t address) {



106  // save address of target and clear buffer



107  txAddress = address;



108  txBuffer.clear();



109



110  transmissionBegun = true;



111}



112



113// Errors:



114//  0 : Success



115//  1 : Data too long



116//  2 : NACK on transmit of address



117//  3 : NACK on transmit of data



118//  4 : Other error



119uint8_t TwoWire::endTransmission(bool stopBit)



120{



121  transmissionBegun = false ;



122



123  // Start I2C transmission



124  if ( !sercom->startTransmissionWIRE( txAddress, WIRE_WRITE_FLAG ) )



125  {



126    sercom->prepareCommandBitsWire(WIRE_MASTER_ACT_STOP);



127    return 2 ;  // Address error



128  }



129



130  // Send all buffer



131  while( txBuffer.available() )



132  {



133    // Trying to send data



134    if ( !sercom->sendDataMasterWIRE( txBuffer.read_char() ) )



135    {



136      sercom->prepareCommandBitsWire(WIRE_MASTER_ACT_STOP);



137      return 3 ;  // Nack or error



138    }



139  }



140  



141  if (stopBit)



142  {



143    sercom->prepareCommandBitsWire(WIRE_MASTER_ACT_STOP);



144  }   



145



146  return 0;



147}



148



149uint8_t TwoWire::endTransmission()



150{



151  return endTransmission(true);



152}

154size_t TwoWire::write(uint8_t ucData)



155{



156  // No writing, without begun transmission or a full buffer



157  if ( !transmissionBegun || txBuffer.isFull() )



158  {



159    return 0 ;



160  }



161



162  txBuffer.store_char( ucData ) ;



163



164  return 1 ;



165}



166



167size_t TwoWire::write(const uint8_t *data, size_t quantity)



168{



169  //Try to store all data



170  for(size_t i = 0; i < quantity; ++i)



171  {



172    //Return the number of data stored, when the buffer is full (if write return 0)



173    if(!write(data[i]))



174      return i;



175  }



176



177  //All data stored



178  return quantity;



179}

212void TwoWire::onService(void)



213{



214  if ( sercom->isSlaveWIRE() )



215  {



216    if(sercom->isStopDetectedWIRE() || 



217        (sercom->isAddressMatch() && sercom->isRestartDetectedWIRE() && !sercom->isMasterReadOperationWIRE())) //Stop or Restart detected



218    {



219      sercom->prepareAckBitWIRE();



220      sercom->prepareCommandBitsWire(0x03);



221



222      //Calling onReceiveCallback, if exists



223      if(onReceiveCallback)



224      {



225        onReceiveCallback(available());



226      }



227      



228      rxBuffer.clear();



229    }



230    else if(sercom->isAddressMatch())  //Address Match



231    {



232      sercom->prepareAckBitWIRE();



233      sercom->prepareCommandBitsWire(0x03);



234



235      if(sercom->isMasterReadOperationWIRE()) //Is a request ?



236      {



237        txBuffer.clear();



238



239        transmissionBegun = true;



240



241        //Calling onRequestCallback, if exists



242        if(onRequestCallback)



243        {



244          onRequestCallback();



245        }



246      }



247    }



248    else if(sercom->isDataReadyWIRE())



249    {



250      if (sercom->isMasterReadOperationWIRE())



251      {



252        uint8_t c = 0xff;



253



254        if( txBuffer.available() ) {



255          c = txBuffer.read_char();



256        }



257



258        transmissionBegun = sercom->sendDataSlaveWIRE(c);



259      } else { //Received data



260        if (rxBuffer.isFull()) {



261          sercom->prepareNackBitWIRE(); 



262        } else {



263          //Store data



264          rxBuffer.store_char(sercom->readDataWIRE());



265



266          sercom->prepareAckBitWIRE(); 



267        }



268



269        sercom->prepareCommandBitsWire(0x03);



270      }



271    }



272  }



273}
