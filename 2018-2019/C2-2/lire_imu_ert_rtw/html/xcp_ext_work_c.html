27/* Function ====================================================================



28 * Pause the process (w/o hogging the cpu) until the XCP Master has modified the



29 * xcpModelStartRequest value to true or until we are no longer in the paused state.  



30 * The XCP stack must continue to process events (otherwise the host would not 



31 * be able to communicate with the target).



32 */



33void rtExtModePauseIfNeeded(RTWExtModeInfo *ei,



34    int_T          numSampTimes,



35    boolean_T      *stopReqPtr)



36{



37    XCP_UNUSED_PARAM(ei);



38    XCP_UNUSED_PARAM(numSampTimes);



39



40    if (stopReqPtr == NULL) {



41        XCP_PRINTF("rtExtModePauseIfNeeded error: invalid stopReqPtr\n");



42        return;



43    }



44



45    while ((xcpModelStatus == TARGET_STATUS_PAUSED) && !xcpModelStartRequest && !(*stopReqPtr)) {



46        XCP_SLEEP(0, EXTMODE_RETRY_TIME_IN_MICROSECONDS);



47        xcpExtModeRunBackground(DEFAULT_XCP_EXTMODE_RUN_BACKGROUND_FLUSH);



48        *stopReqPtr = xcpModelStopRequest;



49    }



50



51    if (xcpModelStatus == TARGET_STATUS_RUNNING) {



52        /* Add a dummy delay, that forces the task to pause the execution 



53           (leaving the 'running' state). This gives Simulink the opportunity 



54           to run when connecting to host-based targets in External Mode */



55          XCP_SLEEP(0, 0);



56    }



57



58    xcpModelStartRequest = false; /* reset to false - if we were stepped we want to



59                                   *                  stop again next time we get



60                                   *                  back here.



61                                   */



62} /* end rtExtModePauseIfNeeded */

64/* Function ====================================================================



65 * Pause the process (w/o hogging the cpu) until the XCP Master has modified the



66 * xcpModelStart value to true.  



67 * The XCP stack must continue to process events (otherwise the host would not 



68 * be able to communicate with the target)



69 */



70void rtExtModeWaitForStartPkt(RTWExtModeInfo *ei,



71                              int_T          numSampTimes,



72                              boolean_T      *stopReqPtr)



73{



74    XCP_UNUSED_PARAM(numSampTimes);



75



76    if (stopReqPtr == NULL) {



77        XCP_PRINTF("rtExtModeWaitForStartPkt error: invalid stopReqPtr\n");



78        return;



79    }



80



81    /* Initialize the model checksum information */



82    xcpModelChecksum0 = rteiGetChecksum0(ei);



83    xcpModelChecksum1 = rteiGetChecksum1(ei);



84    xcpModelChecksum2 = rteiGetChecksum2(ei);



85    xcpModelChecksum3 = rteiGetChecksum3(ei);



86



87#if defined(INTEGER_CODE) && INTEGER_CODE == 1



88    xcpModelIntegerCode = 1;



89#else



90    xcpModelIntegerCode = 0;



91#endif



92



93    /*



94     * Pause until the XCP master modifies the xcpModelStartRequest value



95     */



96    while(!xcpModelStartRequest && !(*stopReqPtr)) {



97        XCP_SLEEP(0L, EXTMODE_RETRY_TIME_IN_MICROSECONDS);



98        xcpExtModeRunBackground(DEFAULT_XCP_EXTMODE_RUN_BACKGROUND_FLUSH);



99        *stopReqPtr = xcpModelStopRequest;



100    }



101    if (xcpModelStatus != TARGET_STATUS_PAUSED) {



102        xcpModelStatus = TARGET_STATUS_RUNNING;



103    } else {



104        /* leave in pause mode */



105    }



106}

108void rtExtModeOneStep(RTWExtModeInfo *ei,



109                      int_T          numSampTimes,



110                      boolean_T      *stopReqPtr)



111{



112    UNUSED_PARAMETER(ei);



113    UNUSED_PARAMETER(numSampTimes);



114



115    if (stopReqPtr == NULL) {



116        XCP_PRINTF("rtExtModeOneStep error: invalid stopReqPtr\n");



117        return;



118    }



119



120    /*



121     * Call the XCP background task



122     */



123    if (xcpModelStatus != TARGET_STATUS_PAUSED) {



124        xcpExtModeRunBackground(DEFAULT_XCP_EXTMODE_RUN_BACKGROUND_FLUSH);



125        *stopReqPtr = xcpModelStopRequest;



126    }



127}

129void rtExtModeUpload(int_T tid, real_T taskTime)



130{



131    extmodeEventId_T eventId = 0;



132    extmodeSimulationTime_T simulationTime = (extmodeSimulationTime_T)  taskTime;



133



134    if ((tid < 0) || (tid > EXTMODE_MAX_EVENT_ID)) {



135        XCP_PRINTF("rtExtModeUpload error: invalid tid (%d)\n", tid);



136        return;



137    }



138    



139    eventId = (extmodeEventId_T) tid;



140



141    xcpExtmodeEvent(eventId, simulationTime);



142}

174void rtExtModeParseArgs(int_T        argc, 



175                        const char_T *argv[],



176                        real_T       *unused)



177{



178    UNUSED_PARAMETER(unused);



179



180#ifdef  EXTMODE_DISABLE_ARGS_PROCESSING



181



182    /* Some targets do not support command line args */



183    UNUSED_PARAMETER(argc);



184    UNUSED_PARAMETER(argv);



185



186#else   



187



188    {/* Extract the "-w" option (wait for packet from host), if present.



189        Note: in Classic External mode this parameter is handled within rtiostream_interface.c,



190        but this file is no longer present in XCP-based External Mode.



191        Let all the remaining parameters pass through to XCP-based external mode for parsing.  



192        NULL out the "-w" argument so that it can then be ignored once handled. */



193        int_T  count = 1;



194



195        while (count < argc) {



196            const char_T *option = argv[count++];



197



198            if ((option != NULL) && (strcmp(option, "-w") == 0)) {



199                xcpModelStartRequest = false;



200                argv[count - 1] = NULL;



201            }



202        }



203    }



204



205    /* Extract XCP specific initialization parameters */



206    xcpExtModeParseArgs(argc, argv);



207



208#endif  /* EXTMODE_DISABLE_ARGS_PROCESSING */



209}

225}

239void rtParseArgsForExtMode(int_T        argc, 



240                           const char_T *argv[])



241{



242#ifdef  EXTMODE_DISABLE_ARGS_PROCESSING



243



244    /* Some targets do not support command line args */



245    UNUSED_PARAMETER(argc);



246    UNUSED_PARAMETER(argv);



247



248#else    



249



250    /* parse command line args */



251    int_T  oldStyle_argc;



252    const char_T *oldStyle_argv[5];



253



254    if ((argc > 1) && (argv[1][0] != '-')) {



255        /* old style */



256        if ( argc > 3 ) {



257            displayUsage();



258            exit(EXIT_FAILURE);



259        }



260



261        oldStyle_argc    = 1;



262        oldStyle_argv[0] = argv[0];



263    



264        if (argc >= 2) {



265            oldStyle_argc = 3;



266



267            oldStyle_argv[1] = "-tf";



268            oldStyle_argv[2] = argv[1];



269        }



270



271        if (argc == 3) {



272            oldStyle_argc = 5;



273



274            oldStyle_argv[3] = "-port";



275            oldStyle_argv[4] = argv[2];



276        }



277



278        argc = oldStyle_argc;



279        argv = oldStyle_argv;



280    }



281



282    {



283        /* new style: */



284    	extmodeSimulationTime_T  timeValue = 0;



285        char_T tmpStr2[200];



286        int_T  count      = 1;



287        int_T  parseError = false;



288



289        /*



290         * Parse the standard RTW parameters.  Let all unrecognized parameters



291         * pass through to external mode for parsing.  NULL out all args handled



292         * so that the external mode parsing can ignore them.



293         */



294        while (count < argc) {



295            const char_T *option = argv[count++];



296



297            if (option != NULL) {



298                /* final time */



299                if ((strcmp(option, "-tf") == 0) && (count != argc)) {



300                    const char_T *tfStr = argv[count++];



301



302                    sscanf(tfStr, "%200s", tmpStr2);



303                    if (strcmp(tmpStr2, "inf") == 0) {



304                        timeValue = EXTMODE_SIMULATION_RUN_FOREVER;



305                    }



306                    else {



307                        char_T tmpstr[2];



308#if defined(INTEGER_CODE) && INTEGER_CODE == 1



309                        int tmpValue;



310



311                        if ((sscanf(tmpStr2, "%d%1s", &tmpValue, tmpstr) != 1) ||



312                            (tmpValue < 0)) {



313                            XCP_PRINTF("External mode final simulation time must be a positive, integer value or inf\n");



314                            parseError = true;



315                            break;



316                        }



317                        timeValue = (extmodeSimulationTime_T) tmpValue;



318#else



319                        if ((sscanf(tmpStr2, "%lf%1s", &timeValue, tmpstr) != 1) ||



320                            (timeValue < (real_T)0)) {



321                            XCP_PRINTF("External mode final simulation time must be a positive, real value or inf\n");



322                            parseError = true;



323                            break;



324                        }



325#endif



326                    }



327                    xcpExtmodeFinalSimulationTime = timeValue;



328



329                    argv[count - 2] = NULL;



330                    argv[count - 1] = NULL;



331                }



332                /* -w (wait for packet from host) option */



333                else if (strcmp(option, "-w") == 0) {



334                    xcpModelStartRequest = false;



335                    argv[count - 1] = NULL;



336                }



337            }



338        }



339



340        if (parseError) {



341            XCP_PRINTF("\nUsage: model_name -option1 val1 -option2 val2 -option3 "



342                       "...\n\n");



343#if defined(INTEGER_CODE) && INTEGER_CODE == 1



344            XCP_PRINTF("\t-tf 20 - sets final time to 20 ticks\n");



345#else



346            XCP_PRINTF("\t-tf 20 - sets final time to 20 seconds\n");



347#endif



348            exit(EXIT_FAILURE);



349        }



350



351        /* Extract XCP specific initialization parameters */



352        xcpExtModeParseArgs(argc, argv);



353



354        /*



355         * Check for unprocessed ("unhandled") args.



356         */



357        {



358            int i;



359            for (i=1; i<argc; i++) {



360                if (argv[i] != NULL) {



361                    XCP_PRINTF("Unexpected command line argument: %s\n", argv[i]);



362                    exit(EXIT_FAILURE);



363                }



364            }



365        }



366    }



367



368    if (xcpExtmodeFinalSimulationTime == EXTMODE_SIMULATION_RUN_FOREVER) {



369        XCP_PRINTF("\n**warning: the simulation will run with no stop time due "



370                   "to external mode with '-tf inf' argument.\n");



371    }



372



373#endif /* EXTMODE_DISABLE_ARGS_PROCESSING */



374}
