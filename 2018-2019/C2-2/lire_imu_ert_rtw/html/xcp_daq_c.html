51static boolean_T isActiveDaqList(uint16_T daqListId)



52{



53    return (xcpDynamicDaqLists.daq[daqListId].status == XCP_DAQ_SELECTED) ||



54           (xcpDynamicDaqLists.daq[daqListId].status == XCP_DAQ_STARTED);



55}

57static boolean_T noActiveDaqLists(void)



58{



59    boolean_T found = 0;



60    uint16_T i = 0;



61



62    for (i = 0; (i < xcpDynamicDaqLists.daqCount) && !found; i++) {



63        found = isActiveDaqList(i);



64    }



65



66    return !found;



67}

69static boolean_T isValidDaqPtr(uint16_T daqListId, uint8_T odtId, uint8_T odtEntryId)



70{



71    boolean_T ret = (xcpDynamicDaqLists.daq != NULL) &&



72                    (daqListId < xcpDynamicDaqLists.daqCount) &&



73                    (xcpDynamicDaqLists.daq[daqListId].odt != NULL) &&



74                    (odtId < xcpDynamicDaqLists.daq[daqListId].odtCount) &&



75                    (xcpDynamicDaqLists.daq[daqListId].odt[odtId].entry != NULL) &&



76                    (odtEntryId < xcpDynamicDaqLists.daq[daqListId].odt[odtId].entriesCount);



77



78    return ret;



79}

81static boolean_T isValidDaqEntry(uint8_T bitOffset, uint8_T size, uint32_T address)



82{



83    boolean_T ret = false;



84    uint16_T  entrySize = XCP_IN_BYTES(size);



85



86    if (bitOffset == 0xFF) {



87        /* "Normal" (non-bitwise) access has been selected */



88        if ( (entrySize <= XCP_MAX_ODT_ENTRY_SIZE) &&



89             (entrySize % (XCP_ODT_ENTRY_SIZE_GRANULARITY/XCP_ADDRESS_GRANULARITY_BYTES_NUMBER) == 0) &&



90             (  address % (XCP_ODT_ENTRY_SIZE_GRANULARITY/XCP_ADDRESS_GRANULARITY_BYTES_NUMBER) == 0) ) {



91            ret = true;



92        }



93    }



94    else {



95        /* Bitwise access has been selected */



96        if ((bitOffset <= 0x1F) && (address % XCP_ODT_ENTRY_SIZE_GRANULARITY) &&



97            (entrySize == XCP_ODT_ENTRY_SIZE_GRANULARITY)) {



98            ret = true;



99        }



100    }



101



102    return ret;



103}

105/** The current supported mode includes:



106- alternating = 0



107- direction = 0 (DAQ only)



108- timestamp = 1 or 0



109- PID OFF = 0



110*/



111static boolean_T isSupportedDaqListMode(uint8_T mode)



112{



113    #if XCP_TIMESTAMP_FIXED == 0



114        return ( (mode & ~XCP_DAQ_MODE_TIMESTAMP_MASK) == 0 );



115    #else



116        return (mode == XCP_DAQ_MODE_TIMESTAMP_MASK);



117    #endif



118}

120static boolean_T isValidStartStopMode(uint8_T mode)



121{



122    boolean_T ret = false;



123



124    if ((mode == XCP_DAQ_LIST_STOP) ||



125        (mode == XCP_DAQ_LIST_START) ||



126        (mode == XCP_DAQ_LIST_SELECT))



127        ret = true;



128



129    return ret;

149/* Try to allocate a reserved memory pool dedicated to the packets in the DAQ list.



150   @note depending on the configuration of the memory allocator, this may fail



151         so we just set daq->poolId to XCP_INVALID_POOL_ID in that case */



152static XcpErrorCode createDaqReservedPool(XcpDaq *daq)



153{



154    XcpErrorCode errorCode = XCP_SUCCESS;



155    size_t poolBlockSize = 0;



156    int i = 0;



157



158    /* Calculate the max size for the ODT packets in the given DAQ list */



159    for (i = 0; i < daq->odtCount; i++) {



160        /* Calculate the required message buffer size */



161        size_t packetSize = XCP_IN_BYTES(daq->odt[i].size + XCP_IN_AG(XCP_ID_FIELD_SIZE));



162        size_t msgBufferSize = 0;



163



164        if (i == 0) {



165            /* The first DTO should also contain the Timestamp info */



166            packetSize += XCP_TIMESTAMP_SIZE;



167        }



168



169        msgBufferSize = xcpTransportMsgBufferSize(packetSize, XCP_DTO);



170



171        if (msgBufferSize > poolBlockSize) {



172            poolBlockSize = msgBufferSize;



173        }



174    }



175



176    /* Try the allocation of the dedicate pool */



177    errorCode = xcpMemReservedPoolCreate(poolBlockSize,



178    		                             XCP_MEM_DAQ_RESERVED_POOL_BLOCKS_NUMBER,



179                                         &daq->poolId);



180    if (errorCode != XCP_SUCCESS) {



181        daq->poolId = XCP_INVALID_POOL_ID;



182    }



183



184    return errorCode;



185}



186



187static void destroyDaqReservedPool(XcpDaq *daq)



188{



189    if (daq->poolId != XCP_INVALID_POOL_ID) {



190        xcpMemReservedPoolDestroy(daq->poolId);



191    }



192



193    daq->poolId = XCP_INVALID_POOL_ID;



194}

200static XcpProtoErrorCode setDaqPtrInputPacketHandler(void   *msgBuffer,



201                                                     size_t  xcpPacketOffset,



202                                                     size_t *outputPacketSize)



203{



204    XcpProtoErrorCode protoErrorCode = XCP_PROTO_SUCCESS;



205    uint8_T *packet = (uint8_T *)msgBuffer + xcpPacketOffset;



206    XcpSetDaqPtrCmdPacketFrame *frame = (XcpSetDaqPtrCmdPacketFrame *) packet;



207    uint16_T daqListId  = (uint16_T)frame->daqListId;



208    uint8_T  odtId      = (uint8_T)frame->odtId;



209    uint8_T  odtEntryId = (uint8_T)frame->odtEntryId;



210    boolean_T ok = false;



211



212    /* Check if the selected ODT Entry is available */



213    ok = isValidDaqPtr(daqListId, odtId, odtEntryId);



214    XCP_INPUT_PKT_ERROR_IF(!ok, XCP_PROTO_OUT_OF_RANGE,



215        ("SET_DAQ_PTR: ODT entry not available (DAQ %d, ODT %d, Entry %d)\n", daqListId, odtId, odtEntryId));



216



217    /* Check if the DAQ list is currently active */



218    ok = !isActiveDaqList(daqListId);



219    XCP_INPUT_PKT_ERROR_IF(!ok, XCP_PROTO_DAQ_ACTIVE, ("SET_DAQ_PTR: DAQ list %d is currently active\n", daqListId));



220



221    /* Everything is fine -> proceed and update the current ODT entry pointer */



222    XCP_PRINTF("SET_DAQ_PTR: updating current ODT entry pointer to DAQ %d, ODT %d, Entry %d\n",



223               daqListId, odtId, odtEntryId);



224



225    xcpCurrentDaq.daqListId  = daqListId;



226    xcpCurrentDaq.odtId      = odtId;



227    xcpCurrentDaq.odtEntryId = odtEntryId;



228



229    *outputPacketSize = XCP_GENERIC_RES_PACKET_SIZE_IN_BYTES;



230



231    return protoErrorCode;



232}

234/*****************************************************************************



235    XCP WRITE DAQ



236******************************************************************************/



237static XcpProtoErrorCode writeDaqInputPacketHandler(void   *msgBuffer,



238                                                    size_t  xcpPacketOffset,



239                                                    size_t *outputPacketSize)



240{



241    XcpProtoErrorCode protoErrorCode = XCP_PROTO_SUCCESS;



242    uint8_T *packet = (uint8_T *)msgBuffer + xcpPacketOffset;



243    XcpWriteDaqCmdPacketFrame *frame = (XcpWriteDaqCmdPacketFrame *)packet;



244    uint16_T daqListId = xcpCurrentDaq.daqListId;



245    uint8_T  odtId = xcpCurrentDaq.odtId;



246    uint8_T  odtEntryId = xcpCurrentDaq.odtEntryId;



247    XcpOdtEntry *entry = NULL;



248    boolean_T ok = false;



249



250    /* Check if the selected ODT Entry is available */



251    ok = isValidDaqPtr(daqListId, odtId, odtEntryId);



252    XCP_INPUT_PKT_ERROR_IF(!ok, XCP_PROTO_WRITE_PROTECTED,



253        ("WRITE_DAQ: current ODT entry not available for write operations (DAQ %d, ODT %d, Entry %d)\n", daqListId, odtId, odtEntryId));



254



255    /* Check if the DAQ list is currently active */



256    ok = !isActiveDaqList(daqListId);



257    XCP_INPUT_PKT_ERROR_IF(!ok, XCP_PROTO_DAQ_ACTIVE, ("XCP_WRITE_DAQ: DAQ list %d is currently active\n", daqListId));



258



259    /* Check if the Entry values are valid */



260    ok = isValidDaqEntry((uint8_T)frame->bitOffset, (uint8_T)frame->size, (uint32_T)frame->address);



261    XCP_INPUT_PKT_ERROR_IF(!ok, XCP_PROTO_OUT_OF_RANGE,



262        ("XCP_WRITE_DAQ: invalid DAQ %d ODT %d Entry %d value: bitOffset = %xH size = %d addressExtension = %dH address = %xH\n",



263         daqListId, odtId, odtEntryId, frame->bitOffset, frame->size, frame->addressExtension, frame->address));



264



265    /* Everything is fine -> proceed and write the entry */



266    XCP_PRINTF("XCP_WRITE_DAQ: writing DAQ %d ODT %d Entry %d: bitOffset = %xH size = %d addressExtension = %dH address = %xH\n",



267               daqListId, odtId, odtEntryId, frame->bitOffset, frame->size, frame->addressExtension, frame->address);



268



269    entry = &(xcpDynamicDaqLists.daq[daqListId].odt[odtId].entry[odtEntryId]);



270



271    entry->bitOffset        = (uint8_T)frame->bitOffset;



272    entry->size             = (uint8_T)frame->size;



273    entry->addressExtension = (uint8_T)frame->addressExtension;



274    entry->address          = (uint32_T)frame->address;



275



276    /* After a successful WRITE_DAQ, the xcpDaqPtr odtEntryId needs to be post incremented */



277    xcpCurrentDaq.odtEntryId++;



278



279    *outputPacketSize = XCP_GENERIC_RES_PACKET_SIZE_IN_BYTES;



280



281    return protoErrorCode;



282}

287static XcpProtoErrorCode setDaqListModeInputPacketHandler(void   *msgBuffer,



288                                                          size_t  xcpPacketOffset,



289                                                          size_t *outputPacketSize)



290{



291    XcpProtoErrorCode protoErrorCode = XCP_PROTO_SUCCESS;



292    uint8_T *packet = (uint8_T *)msgBuffer + xcpPacketOffset;



293    XcpSetDaqListModeCmdPacketFrame *frame = (XcpSetDaqListModeCmdPacketFrame *)packet;



294    XcpDaq *daqList = NULL;



295    boolean_T ok = false;



296



297    /* Check if the DAQ List is available */



298    ok = (xcpDynamicDaqLists.daq != NULL) && (frame->daqListId < xcpDynamicDaqLists.daqCount);



299    XCP_INPUT_PKT_ERROR_IF(!ok, XCP_PROTO_OUT_OF_RANGE, ("SET_DAQ_LIST_MODE: DAQ list %d not available\n", frame->daqListId));



300



301    daqList = &(xcpDynamicDaqLists.daq[frame->daqListId]);



302



303    /* Check if the DAQ list is currently active */



304    ok = !isActiveDaqList((uint16_T)frame->daqListId);



305    XCP_INPUT_PKT_ERROR_IF(!ok, XCP_PROTO_DAQ_ACTIVE, ("SET_DAQ_LIST_MODE: DAQ list %d is currently active\n", frame->daqListId));



306



307    /* Check if the Event Id is valid */



308    ok = (frame->eventId < XCP_MAX_EVENT_CHANNEL);



309    XCP_INPUT_PKT_ERROR_IF(!ok, XCP_PROTO_OUT_OF_RANGE, ("SET_DAQ_LIST_MODE: invalid event channel Id %d\n", frame->eventId));



310



311    /* Check DAQ list priority */



312    ok = (frame->priority == 0);



313    XCP_INPUT_PKT_ERROR_IF(!ok, XCP_PROTO_OUT_OF_RANGE, ("SET_DAQ_LIST_MODE: priority not supported, must be zero\n"));



314



315    /* Check if DAQ list mode is supported */



316    ok = isSupportedDaqListMode((uint8_T)frame->mode);



317    XCP_INPUT_PKT_ERROR_IF(!ok, XCP_PROTO_MODE_NOT_VALID, ("SET_DAQ_LIST_MODE: DAQ list mode not supported %xH\n", frame->mode));



318



319    /* Everything is fine -> proceed and set the DAQ list mode */



320    XCP_PRINTF("SET_DAQ_LIST_MODE: setting DAQ list %d: mode = %xH, eventId = %d, prescaler = %d, priority = %d\n",



321        frame->daqListId, frame->mode, frame->eventId, frame->prescaler, frame->priority);



322



323    daqList->mode       = (uint8_T)frame->mode;



324    daqList->eventId    = (uint16_T)frame->eventId;



325    daqList->prescaler  = (uint8_T)frame->prescaler;



326    daqList->priority   = (uint8_T)frame->priority;



327



328    /* Assuming that prescaler = 0 means value that you are not interested in using it */



329    if (daqList->prescaler <= 1)



330        daqList->prescaler = 1;



331



332    *outputPacketSize = XCP_GENERIC_RES_PACKET_SIZE_IN_BYTES;



333



334    return protoErrorCode;



335}

337/*****************************************************************************



338    XCP START STOP DAQ LIST



339******************************************************************************/



340static XcpProtoErrorCode startStopDaqListInputPacketHandler(void   *msgBuffer,



341                                                            size_t  xcpPacketOffset,



342                                                            size_t *outputPacketSize)



343{



344    XcpProtoErrorCode protoErrorCode = XCP_PROTO_SUCCESS;



345    uint8_T *packet = (uint8_T *)msgBuffer + xcpPacketOffset;



346    XcpStartStopDaqListCmdPacketFrame *frame = (XcpStartStopDaqListCmdPacketFrame *)packet;



347    XcpDaq *daqList = NULL;



348    int32_T daqListId = frame->daqListId;



349    boolean_T ok = false;



350    boolean_T locked = false;



351



352    /* Check if the DAQ List is available */



353    ok = (xcpDynamicDaqLists.daq != NULL) && (daqListId < xcpDynamicDaqLists.daqCount);



354    XCP_INPUT_PKT_ERROR_IF(!ok, XCP_PROTO_OUT_OF_RANGE,



355        ("START_STOP_DAQ_LIST: DAQ list %d not available\n", daqListId));



356



357    daqList = &(xcpDynamicDaqLists.daq[daqListId]);



358



359    /* Check if requested mode is valid */



360    ok = isValidStartStopMode((uint8_T)frame->mode);



361    XCP_INPUT_PKT_ERROR_IF(!ok, XCP_PROTO_MODE_NOT_VALID,



362        ("START_STOP_DAQ_LIST: invalid mode %d selected \n", frame->mode));



363



364    /* Assign the Absolute ODT number */



365    if (absoluteOdtNumberIdentification()) {



366        if (daqList->status == XCP_DAQ_INIT) {



367            /* The DAQ list has never been started/stopped/selected before



368            -> PIDs for the ODTs needs to be reserved */



369            if (XCP_READ_BIT_VALUE(daqList->mode, XCP_DAQ_MODE_DIRECTION_MASK)) {



370                /* STIM */



371                ok = (daqList->odt != NULL) &&



372                       ((xcpDynamicDaqLists.firstAvailableStimPid +



373                        daqList->odtCount) < XCP_MAX_STIM_ODT_NUMBER);



374



375                if (ok) {



376                    daqList->firstPid = xcpDynamicDaqLists.firstAvailableStimPid;



377                    xcpDynamicDaqLists.firstAvailableStimPid += daqList->odtCount;



378                }



379            }



380            else {



381                /* DAQ */



382                ok = (daqList->odt != NULL) &&



383                       ((xcpDynamicDaqLists.firstAvailableDaqPid +



384                        daqList->odtCount) < XCP_MAX_DAQ_ODT_NUMBER);



385



386                if (ok) {



387                    daqList->firstPid = xcpDynamicDaqLists.firstAvailableDaqPid;



388                    xcpDynamicDaqLists.firstAvailableDaqPid += daqList->odtCount;



389                }



390            }



391



392            XCP_INPUT_PKT_ERROR_IF(!ok, XCP_PROTO_DAQ_CONFIG_ERROR,



393                ("START_STOP_DAQ_LIST: cannot assign PIDs for DAQ list %d \n", daqListId));



394        }



395    }



396



397    /* Check the ODT configuration and also if the ODT sizes fits the Max DTO size */



398    if ((frame->mode == XCP_DAQ_LIST_START) || (frame->mode == XCP_DAQ_LIST_SELECT)) {



399        int i = 0;



400



401        ok = true;



402        for (i = 0; (i < daqList->odtCount) && ok; i++) {



403            int j = 0;



404            size_t odtSize = 0;



405



406            /* Check if we have ODT entries */



407            ok = (daqList->odt[i].entry != NULL) && (daqList->odt[i].entriesCount > 0);



408            XCP_INPUT_PKT_ERROR_IF(!ok, XCP_PROTO_DAQ_CONFIG_ERROR,



409                ("START_STOP_DAQ_LIST: wrong ODT entry configuration detected for DAQ list %d, ODT %d \n", daqListId, i));



410



411            /* Calculate the resulting ODT size */



412            for (j = 0; j < daqList->odt[i].entriesCount; j++) {



413                odtSize += daqList->odt[i].entry[j].size;



414            }



415



416            /* Check that the resulting ODT size is valid */



417            ok = (XCP_IN_BYTES(odtSize + XCP_IN_AG(XCP_ID_FIELD_SIZE)) + XCP_TIMESTAMP_SIZE) <= xcpTransportMaxDtoSize();



418            XCP_INPUT_PKT_ERROR_IF(!ok, XCP_PROTO_DAQ_CONFIG_ERROR,



419                ("START_STOP_DAQ_LIST: in DAQ list %d the ODT entries size is larger than Max DTO size for ODT %d \n", daqListId, i));



420



421            /* Update the ODT size */



422            daqList->odt[i].size = odtSize;



423        }



424    }



425



426    /* Everything is fine -> proceed and try to change the DAQ list status */



427    XCP_WRITE_TRY_LOCK(xcpDaqLock, xcpRunningEventCounter, locked);



428



429    if (locked) {



430    	XcpErrorCode createPoolError = XCP_SUCCESS;



431



432        switch (frame->mode) {



433        case XCP_DAQ_LIST_STOP:



434            XCP_PRINTF("START_STOP_DAQ_LIST: stopping DAQ list %d\n", daqListId);



435            daqList->status = XCP_DAQ_STOPPED;



436            if (noActiveDaqLists()) {



437                /* update status if this was the only active DAQ list */



438                xcpStatusSet(XCP_CONNECTED);



439                xcpSessionStatusClearMask(XCP_SESSION_DAQ_RUNNING_MASK);



440            }



441            break;



442        case XCP_DAQ_LIST_START:



443            XCP_PRINTF("START_STOP_DAQ_LIST: starting DAQ list %d\n", daqListId);



444



445            /* daqList->poolId is initialized with XCP_INVALID_POOL_ID in 



446               allocDaqInputPacketHandler.



447



448               Allocate a reserved memory pool for the DAQ list, if needed. */



449            if ((daqListId < XCP_MEM_DAQ_RESERVED_POOLS_NUMBER) && 



450                (daqList->poolId == XCP_INVALID_POOL_ID)) {



451                /* Create a reserved memory pool, if not already created */



452                createPoolError = createDaqReservedPool(daqList);



453            }



454



455            if (createPoolError == XCP_SUCCESS) {



456                daqList->status = XCP_DAQ_STARTED;



457                xcpStatusSet(XCP_SYNC_DATA_TRANSFER);



458                xcpSessionStatusSetMask(XCP_SESSION_DAQ_RUNNING_MASK);



459            }



460            break;



461        case XCP_DAQ_LIST_SELECT:



462            XCP_PRINTF("START_STOP_DAQ_LIST: selected DAQ list %d\n", frame->daqListId);



463            daqList->status = XCP_DAQ_SELECTED;



464            break;



465        }



466



467        XCP_WRITE_UNLOCK(xcpDaqLock);



468



469        *outputPacketSize = XCP_START_STOP_DAQ_LIST_RES_PACKET_SIZE_IN_BYTES;



470        XCP_INPUT_PKT_ERROR_IF(createPoolError != XCP_SUCCESS, XCP_PROTO_MEMORY_OVERFLOW,



471                ("START_STOP_DAQ_LIST: cannot allocate reserved memory pool for the DAQ list\n"));



472        



473    }



474



475    /* Error if the command cannot be executed as the processing of



476       at least one event was in progress */



477    XCP_INPUT_PKT_ERROR_IF(!locked, XCP_PROTO_BUSY,



478        ("START_STOP_DAQ_LIST: cannot be carried out as an event was processed\n"));



479



480    return protoErrorCode;



481}

499/*****************************************************************************



500    XCP START STOP SYNCH



501******************************************************************************/



502static XcpProtoErrorCode startStopSynchInputPacketHandler(void   *msgBuffer,



503                                                          size_t  xcpPacketOffset,



504                                                          size_t *outputPacketSize)



505{



506    XcpProtoErrorCode protoErrorCode = XCP_PROTO_SUCCESS;



507    uint8_T *packet = (uint8_T *)msgBuffer + xcpPacketOffset;



508    XcpStartStopSynchCmdPacketFrame *frame = (XcpStartStopSynchCmdPacketFrame *)packet;



509    boolean_T ok = false;



510    boolean_T locked = false;



511    int32_T i = 0;



512



513    /* Check if requested mode is valid */



514    ok = isValidStartStopSynchMode((uint8_T)frame->mode);



515    XCP_INPUT_PKT_ERROR_IF(!ok, XCP_PROTO_MODE_NOT_VALID,



516        ("START_STOP_SYNCH: invalid mode %d selected \n", frame->mode));



517



518    /* Check if the DAQ Lists are available */



519    ok = (xcpDynamicDaqLists.daq != NULL) && (xcpDynamicDaqLists.daqCount > 0);



520    XCP_INPUT_PKT_ERROR_IF(!ok, XCP_PROTO_OUT_OF_RANGE,



521        ("START_STOP_SYNCH: DAQ lists not available\n"));



522



523    /* Everything is fine -> proceed and try to change the DAQ list status */



524



525    /* Need to get the lock to guarantee that they are all done in the same cycle */



526    XCP_WRITE_TRY_LOCK(xcpDaqLock, xcpRunningEventCounter, locked);



527



528    if (locked) {



529        XcpErrorCode createPoolError = XCP_SUCCESS;



530        uint16_T daqListId = 0;



531



532        switch (frame->mode) {



533        case XCP_DAQ_LIST_STOP_ALL:



534            for (i = 0; i < xcpDynamicDaqLists.daqCount; i++) {



535                if (xcpDynamicDaqLists.daq[i].status != XCP_DAQ_INIT) {



536                    XCP_PRINTF("START_STOP_SYNCH: stopping DAQ list %d\n", i);



537                    xcpDynamicDaqLists.daq[i].status = XCP_DAQ_STOPPED;



538                }



539            }



540



541            /* update status as there are no more active DAQ list */



542            xcpStatusSet(XCP_CONNECTED);



543            xcpSessionStatusClearMask(XCP_SESSION_DAQ_RUNNING_MASK);



544            break;



545



546        case XCP_DAQ_LIST_START_SELECTED:



547            /* xcpDynamicDaqLists.daq[i].poolId is initialized with



548               XCP_INVALID_POOL_ID in allocDaqInputPacketHandler.



549



550               Allocate a reserved memory pool for the DAQ list, if needed. */



551            for (i = 0; (i < xcpDynamicDaqLists.daqCount) && (createPoolError == XCP_SUCCESS); i++) {



552                if (xcpDynamicDaqLists.daq[i].status == XCP_DAQ_SELECTED) {



553                    if ((i < XCP_MEM_DAQ_RESERVED_POOLS_NUMBER) && 



554                        (xcpDynamicDaqLists.daq[i].poolId == XCP_INVALID_POOL_ID)) {



555                        /* Create a reserved memory pool, if not already created */



556                        createPoolError = createDaqReservedPool(&xcpDynamicDaqLists.daq[i]);



557                        daqListId = (uint16_T) i;



558                    }



559                }



560            }



561



562            if (createPoolError == XCP_SUCCESS) {



563                /* If the allocation went well for all the DAQ lists, start them */



564                for (i = 0; (i < xcpDynamicDaqLists.daqCount); i++) {



565                    if (xcpDynamicDaqLists.daq[i].status == XCP_DAQ_SELECTED) {



566                        XCP_PRINTF("START_STOP_SYNCH: starting DAQ list %d\n", i);



567



568                        xcpDynamicDaqLists.daq[i].status = XCP_DAQ_STARTED;



569                        xcpStatusSet(XCP_SYNC_DATA_TRANSFER);



570                        xcpSessionStatusSetMask(XCP_SESSION_DAQ_RUNNING_MASK);



571                    }



572                }



573            }



574            break;



575



576        case XCP_DAQ_LIST_STOP_SELECTED:



577            for (i = 0; i < xcpDynamicDaqLists.daqCount; i++) {



578                if (xcpDynamicDaqLists.daq[i].status == XCP_DAQ_SELECTED) {



579                    XCP_PRINTF("START_STOP_SYNCH: stopping DAQ list %d\n", i);



580                    xcpDynamicDaqLists.daq[i].status = XCP_DAQ_STOPPED;



581                }



582            }



583



584            if (noActiveDaqLists()) {



585                /* update status if this was the only active DAQ list */



586                xcpStatusSet(XCP_CONNECTED);



587                xcpSessionStatusClearMask(XCP_SESSION_DAQ_RUNNING_MASK);



588            }



589            break;



590        }



591



592        XCP_WRITE_UNLOCK(xcpDaqLock);



593



594        XCP_INPUT_PKT_ERROR_IF(createPoolError != XCP_SUCCESS, XCP_PROTO_MEMORY_OVERFLOW,



595                ("START_STOP_SYNCH: cannot allocate reserved memory pool for the DAQ list %d\n", daqListId));



596



597        *outputPacketSize = XCP_GENERIC_RES_PACKET_SIZE_IN_BYTES;



598    }



599



600    /* Error if the command cannot be executed as the processing of 



601       at least one event was in progress */



602    XCP_INPUT_PKT_ERROR_IF(!locked, XCP_PROTO_BUSY,



603        ("START_STOP_SYNCH: cannot be carried out as an event was processed\n"));



604



605    return protoErrorCode;



606}

892XcpErrorCode xcpDaqEvent(XcpEventIdType eventId, uint32_T timestamp)



893{



894    XcpErrorCode errorCode = XCP_SUCCESS;



895    boolean_T error = false;



896



897    /* The lock is used to protect against the concurrent execution of XCP commands



898    (within xcpRun()) that can modify the DAQ list data structures */



899    XCP_READ_LOCK(xcpDaqLock, xcpRunningEventCounter);



900



901    if ((xcpDynamicDaqLists.daq != NULL) && (xcpDynamicDaqLists.daqCount > 0)) {



902        int i = 0;



903        XcpDaq *daq = xcpDynamicDaqLists.daq;



904



905        for (i = 0; (i < xcpDynamicDaqLists.daqCount) && !error; i++) {



906            if (!XCP_READ_BIT_VALUE(daq[i].mode, XCP_DAQ_MODE_DIRECTION_MASK) && /* it's a DAQ list */



907                (daq[i].status == XCP_DAQ_STARTED) &&                            /* it's running */



908                (daq[i].eventId == eventId)) {                                   /* and associated to this event */



909                int j = 0;



910



911                /* @note prescaler is not supported at the moment -> we send DTOs every time */



912                for (j = 0; (j < daq[i].odtCount) && !error; j++) {



913                    /* Calculate the required message buffer size */



914                    size_t packetOffset = xcpTransportPacketOffset();



915                    size_t packetSize = 0;



916                    uint8_T *msgBuffer = NULL;



917                    size_t msgBufferSize = 0;



918                    const boolean_T needsTimestamp = IS_TIMESTAMP_REQUIRED(daq[i].mode, j);



919



920                    packetSize = XCP_IN_BYTES(daq[i].odt[j].size + XCP_IN_AG(XCP_ID_FIELD_SIZE));



921                    if (needsTimestamp) {



922                        /* The first DTO should also contain the Timestamp info */



923                        packetSize += XCP_TIMESTAMP_SIZE;



924                    }



925



926                    /* Allocate the message buffer */



927                    msgBufferSize = xcpTransportMsgBufferSize(packetSize, XCP_DTO);



928



929                    if (daq[i].poolId != XCP_INVALID_POOL_ID) {



930                        /* A dedicated memory pool is available */



931                        msgBuffer = (uint8_T *)xcpMemAllocFromPool(daq[i].poolId, msgBufferSize);



932                    } else {



933                        /* No memory pool available, use the Main memory instead */



934                        msgBuffer = (uint8_T *)xcpMemAlloc(msgBufferSize);



935                    }



936



937                    if (msgBuffer == NULL) {



938                        error = true;



939                        errorCode = XCP_NO_MEMORY;



940                    }



941



942                    if (!error) {



943                        uint8_T *packet = msgBuffer + packetOffset;



944                        int k = 0;



945



946                        /* Fill the Identification Field first



947                        @note this is hard-coded to support absolute ODT number at the moment */



948                        *packet = (uint8_T)(daq[i].firstPid + j);



949                        packet++;



950



951                        /* Add the timestamp for the first DTO



952                        @note this is hard-coded to support timestamp size of 4 bytes */



953                        if (needsTimestamp) {



954                            XCP_MEMCPY(packet, &timestamp, sizeof(timestamp));



955



956                            packet += sizeof(timestamp);



957                        }



958



959                        /* Finally copy the ODT entries values */



960                        for (k = 0; k < daq[i].odt[j].entriesCount; k++) {



961                            XcpOdtEntry *entry = &(daq[i].odt[j].entry[k]);



962                            uint8_T *address = XCP_ADDRESS_GET(entry->addressExtension, entry->address);



963



964                            if (daq[i].odt[j].entry[k].bitOffset != 0xFF) {



965                                /* Bitwise access has been selected */



966                                uint32_T *dword = (uint32_T *)address;



967



968                                *packet = (uint8_T) (*dword & ((uint32_T)(1 << entry->bitOffset))) != 0;



969                            }



970                            else {



971                                /* 'Normal' (non-bitwise) access has been selected */



972                                XCP_MEMCPY(packet, address, entry->size);



973                            }



974                            packet += entry->size;



975                        }



976



977                        /* Send the response back to the Transport Layer */



978                        errorCode = xcpTransportTxPacketSet(XCP_TRANSPORT_TX_DEFAULT_QUEUE,



979                                                            msgBuffer, msgBufferSize,



980                                                            packetOffset, packetSize);



981                        if (errorCode != XCP_SUCCESS) {



982                            xcpMemFree(msgBuffer);



983                        }



984                    }



985                }



986            }



987        }



988    }



989



990    XCP_READ_UNLOCK(xcpDaqLock, xcpRunningEventCounter);



991



992    return errorCode;



993}

995boolean_T xcpResetDaqListStatus(void)



996{



997    boolean_T locked = true;



998



999    if (xcpDynamicDaqLists.daq != NULL) {



1000        XcpDaq *daq = xcpDynamicDaqLists.daq;



1001        int32_T  daqCount = xcpDynamicDaqLists.daqCount;



1002



1003        XCP_WRITE_TRY_LOCK(xcpDaqLock, xcpRunningEventCounter, locked);



1004



1005        if (locked) {



1006            /* No events were processed, it's safe to remove the DAQ lists */



1007            int32_T i = 0;



1008            int32_T j = 0;



1009



1010            xcpDynamicDaqLists.daq = NULL;



1011            xcpDynamicDaqLists.daqCount = 0;



1012            xcpDynamicDaqLists.firstAvailableDaqPid = 0;



1013            xcpDynamicDaqLists.firstAvailableStimPid = 0;



1014



1015            XCP_WRITE_UNLOCK(xcpDaqLock);



1016



1017            /* Prepare the Transport Layer for subsequent connections 



1018              (by resetting the Frame Handler counters and deleting the 



1019              existing packets in TX/RX fifos) */



1020            xcpTransportRestart();



1021



1022            /* Delete all the Dynamic DAQ data structures and 



1023               reserved  memory pools */



1024            for (i = 0; i < daqCount; i++) {



1025                if (i < XCP_MEM_DAQ_RESERVED_POOLS_NUMBER) {



1026                    destroyDaqReservedPool(&daq[i]);



1027                }                



1028                



1029                if (daq[i].odt != NULL) {



1030                    for (j = 0; j < daq[i].odtCount; j++) {



1031                        if (daq[i].odt[j].entry != NULL) {



1032                            xcpMemFree(daq[i].odt[j].entry);



1033                        }



1034                    }



1035



1036                    xcpMemFree(daq[i].odt);



1037                }



1038            }



1039



1040            xcpMemFree(daq);



1041        }



1042



1043        xcpSessionStatusClearMask(XCP_SESSION_DAQ_RUNNING_MASK);



1044    }



1045



1046    return locked;



1047}
